   * UPDATE
   
    UPDATE emp
    SET job = (SELECT job
               FROM emp
               WHERE empno = '7499'),
        deptno = (SELECT deptno
                  FROM dept
                  WHERE dname = 'SALES')
    WHERE empno = 7698;
    
    // 결과
    EMPNO                ENAME      JOB       MGR              HIREDATE              SAL                COMM       DEPTNO
    ------------------  ---------- --------- ----------------- -------------------- ----------------- ----------- -------------
    7698                BLAKE      MANAGER   7839              01/05/01             2850                          30                     
                                                        
                                                              ↓
                                                              
    7698                BLAKE      SALESMAN  7839              01/05/01             2850                          30
